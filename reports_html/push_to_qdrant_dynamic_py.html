def normalize(x: str) -> str:
    if not x:
        return ""
    return " ".join(str(x).lower().strip().split())

# ‚ÄúH√†m n√†y gi√∫p d·ªØ li·ªáu t·ª´ Notion d√π ph·ª©c t·∫°p ƒë·∫øn ƒë√¢u c≈©ng ƒë∆∞·ª£c chu·∫©n h√≥a tr∆∞·ªõc khi embedding.‚Äù
def flatten_recursive(value):
    """
    H√†m ƒë·ªá quy l√†m ph·∫≥ng d·ªØ li·ªáu t·ª´ k·∫øt qu·∫£ Notion API.
    Kh√¥ng c·∫ßn bi·∫øt t√™n c·ªôt, ch·ªâ nh√¨n c·∫•u tr√∫c d·ªØ li·ªáu.
    
    Quy t·∫Øc:
    - Dict c√≥ 'number' -> l·∫•y number
    - Dict c√≥ 'select' -> l·∫•y name
    - Dict c√≥ 'multi_select' -> l·∫•y list name
    - List -> ƒë·ªá quy t·ª´ng ph·∫ßn t·ª≠
    - Chu·ªói JSON (str) b·∫Øt ƒë·∫ßu b·∫±ng '[' ho·∫∑c '{' -> th·ª≠ parse r·ªìi ƒë·ªá quy
    """
    import json
    
    if value is None:
        return None
    if isinstance(value, dict):
        if "type" in value:
            t = value.get("type")
            if t in value: # e.g. "type": "number", "number": ...
                return flatten_recursive(value[t])
        
        # X·ª≠ l√Ω c√°c object c·ª• th·ªÉ
        if "name" in value: # Select, Multi-select value
            return value["name"]
        if "start" in value: # Date
            return value["start"]
        if "email" in value:
            return value["email"]
        if "url" in value:
            return value["url"]
        if "phone_number" in value:
            return value["phone_number"]
        if "number" in value: # Direct number object
            return value["number"]
        if "content" in value: # Text
            return value["content"]
        return str(value)
    if isinstance(value, list):
        return [flatten_recursive(v) for v in value]
    
    # Tr∆∞·ªùng h·ª£p chu·ªói nh∆∞ng l·∫°i l√† JSON (do SQLite l∆∞u JSON text)
    if isinstance(value, str):
        value = value.strip()
        if (value.startswith("{") and value.endswith("}")) or \
           (value.startswith("[") and value.endswith("]")):
            try:
                parsed = json.loads(value)
                return flatten_recursive(parsed)
            except:
                pass # Kh√¥ng ph·∫£i JSON valid, d√πng string g·ªëc
    
    # Gi√° tr·ªã nguy√™n th·ªßy (str, int, float, bool)
    return value
# Nhi·ªám v·ª•:
# L·∫•y m√¥ t·∫£ ng·ªØ nghƒ©a c·ªßa b·∫£ng
# M√¥ t·∫£ n√†y ƒë∆∞·ª£c sinh tr∆∞·ªõc ƒë√≥ b·∫±ng LLM

def get_table_description_from_sqlite(table_name: str) -> str:
    try:
        conn = sqlite3.connect(FAQ_DB_PATH)
        cur = conn.cursor()
        cur.execute("SELECT description FROM collections_config WHERE name=?", (table_name,))
        row = cur.fetchone()
        conn.close()
        return row[0] if row else f"M√¥ t·∫£ b·∫£ng {table_name}"
    except:
        return f"M√¥ t·∫£ b·∫£ng {table_name}"

def get_column_mappings(table_name: str) -> dict:
    """ƒê·ªçc column_mappings t·ª´ collections_config"""
    import json
    try:
        conn = sqlite3.connect(FAQ_DB_PATH)
        cur = conn.cursor()
        cur.execute("SELECT column_mappings FROM collections_config WHERE name=?", (table_name,))
        row = cur.fetchone()
        conn.close()
        
        if row and row[0]:
            return json.loads(row[0])
        return {}
    except Exception as e:
        print(f"  ‚ö†Ô∏è Error reading column mappings: {e}")
        return {}


def get_db_connection():
    return sqlite3.connect(FAQ_DB_PATH)

# Nhi·ªám v·ª•:

# T·∫°o ƒëo·∫°n text ƒë·∫°i di·ªán cho 1 record

# ∆Øu ti√™n c√°c tr∆∞·ªùng quan tr·ªçng (title, question, content‚Ä¶)

# G·∫Øn th√™m t√™n b·∫£ng + mapping ti·∫øng Vi·ªát

# üìå C√¢u n√≥i:

# ‚Äúƒê√¢y l√† b∆∞·ªõc bi·∫øn d·ªØ li·ªáu c√≥ c·∫•u tr√∫c th√†nh ng√¥n ng·ªØ t·ª± nhi√™n ƒë·ªÉ ƒë∆∞a v√†o embedding.‚Äù
def build_embed_text(row_dict: dict, table_name: str, mappings: dict = None) -> str:
    """
    T·∫°o text ƒë·ªÉ embed. ∆Øu ti√™n c√°c tr∆∞·ªùng quan tr·ªçng.
    S·ª≠ d·ª•ng mappings ƒë·ªÉ d·ªãch t√™n c·ªôt sang ti·∫øng Vi·ªát.
    """
    skip_cols = ["notion_id", "last_updated", "approved"]
    priority_cols = [
        "name",
        "title",
        "question",
        "ten",
        "tieu_de",
        "cau_hoi",
        "noidung",
        "content",
        "answer",
    ]

    parts = [f"Ch·ªß ƒë·ªÅ: {table_name}"]

    for col in priority_cols:
        col_lower = col.lower()
        if col_lower in row_dict and row_dict[col_lower]:
            parts.append(str(row_dict[col_lower]))

    for col, value in row_dict.items():
        if col not in skip_cols and col.lower() not in priority_cols and value:
            # Display key: mapped name or original key
            key_display = mappings.get(col, col) if mappings else col
            parts.append(f"{key_display}: {value}")

    return normalize(" ".join(parts))


def row_generator(cursor, batch_size=100):
    """ƒê·ªçc t·ª´ng c·ª•c d·ªØ li·ªáu t·ª´ SQLite, tr√°nh full RAM."""
    while True:
        rows = cursor.fetchmany(batch_size)
        if not rows:
            break
        yield rows


# Nhi·ªám v·ª•:

# L·∫•y danh s√°ch notion_id hi·ªán ƒëang c√≥ trong Qdrant

# L·ªçc theo source_table

# üìå D√πng ƒë·ªÉ:

# Ph√°t hi·ªán record ‚Äúm·ªì c√¥i‚Äù (ƒë√£ xo√° ·ªü SQLite nh∆∞ng c√≤n trong Qdrant)
def get_existing_ids_in_qdrant(client: QdrantClient, table_name: str):
    """
    L·∫•y to√†n b·ªô ID (notion_id) hi·ªán ƒëang c√≥ trong Qdrant cho source_table = table_name.
    D√πng scroll v·ªõi filter.
    """
    existing_ids = set()
    offset = None

    while True:
        points, offset = client.scroll(
            collection_name=GLOBAL_COLLECTION_NAME,
            scroll_filter=Filter(
                must=[
                    FieldCondition(
                        key="source_table",
                        match=MatchValue(value=table_name),
                    )
                ]
            ),
            limit=1000,
            offset=offset,
            with_payload=False,
            with_vectors=False,
        )

        for p in points:
            existing_ids.add(str(p.id))

        if offset is None:
            break

    print(f"  üîé Qdrant currently has {len(existing_ids)} ids for table '{table_name}'")
    return existing_ids

# Nhi·ªám v·ª•:

# Xo√° s·∫°ch d·ªØ li·ªáu c·ªßa 1 b·∫£ng trong Qdrant

# Ch·ªâ d√πng cho b·∫£ng h·ªá th·ªëng (questions_log, conversations‚Ä¶)
def delete_entire_table_from_qdrant(client: QdrantClient, table_name: str):
    """
    X√≥a to√†n b·ªô d·ªØ li·ªáu c·ªßa m·ªôt b·∫£ng kh·ªèi Qdrant (d√πng cho b·∫£ng b·ªã exclude)
    """
    print(f"\n[CLEANUP] Checking excluded table: {table_name}")
    existing_ids = get_existing_ids_in_qdrant(client, table_name)
    
    if not existing_ids:
        print("  ‚úî No data found in Qdrant. Clean.")
        return

    print(f"  üóëÔ∏è Found {len(existing_ids)} items. Deleting...")
    try:
        client.delete(
            collection_name=GLOBAL_COLLECTION_NAME,
            points_selector=list(existing_ids),
        )
        print("  ‚úÖ Deleted all items.")
    except Exception as e:
        print(f"  ‚ö†Ô∏è Error deleting items: {e}")

def sync_table_to_global_collection(table_name: str, embed_model, client: QdrantClient):
    """
    Incremental sync cho 1 b·∫£ng:

    1. ƒê·ªçc d·ªØ li·ªáu hi·ªán t·∫°i t·ª´ SQLite:
        - N·∫øu c√≥ c·ªôt approved & b·∫£ng != 'nganh' ‚Üí ch·ªâ l·∫•y approved = 1
        - Ng∆∞·ª£c l·∫°i ‚Üí l·∫•y t·∫•t c·∫£.
    2. Upsert embedding cho t·∫•t c·∫£ row ƒë√≥ v√†o Qdrant.
    3. L·∫•y danh s√°ch ID ƒëang c√≥ trong Qdrant (theo source_table).
    4. Xo√° c√°c ID c√≥ trong Qdrant nh∆∞ng kh√¥ng c√≤n trong SQLite.
    """

    print(f"\n[SYNC] Processing table: {table_name.upper()}")

    # 1. L·∫•y danh s√°ch ID hi·ªán ƒëang c√≥ trong Qdrant cho b·∫£ng n√†y
    existing_ids = get_existing_ids_in_qdrant(client, table_name)

    # 2. ƒê·ªçc d·ªØ li·ªáu t·ª´ SQLite
    conn = get_db_connection()
    cur = conn.cursor()

    try:
        cur.execute(f"PRAGMA table_info({table_name})")
        columns_info = cur.fetchall()
        if not columns_info:
            print(f"  ‚ùå Table '{table_name}' not found in SQLite.")
            conn.close()
            return
        columns = [col[1] for col in columns_info]
    except Exception as e:
        print(f"  ‚ùå Error reading table info for '{table_name}': {e}")
        conn.close()
        return

    # Load mappings
    mappings = get_column_mappings(table_name)
    if mappings:
        print(f"  üó∫Ô∏è  Loaded {len(mappings)} column mappings for embedding.")

    lower_columns = [c.lower() for c in columns]
    has_approved = "approved" in lower_columns
    lower_table_name = table_name.lower()

    if has_approved and lower_table_name != "nganh":
        sql_query = f"SELECT * FROM {table_name} WHERE approved = 1"
    else:
        sql_query = f"SELECT * FROM {table_name}"

    print(f"  üîé SQL: {sql_query}")
    try:
        cur.execute(sql_query)
    except Exception as e:
        print(f"  ‚ùå Error executing query on '{table_name}': {e}")
        conn.close()
        return

    total_synced = 0
    points_buffer = []
    sqlite_ids = set()  # l∆∞u l·∫°i to√†n b·ªô notion_id hi·ªán c√≥ trong SQLite cho b·∫£ng n√†y

    for rows_chunk in row_generator(cur, batch_size=BATCH_SIZE):
        texts_to_embed = []
        payloads = []
        ids = []

        for row in rows_chunk:
            row_dict = dict(zip(columns, row))
            notion_id = row_dict.get("notion_id")

            if not notion_id:
                continue

            notion_id_str = str(notion_id)
            sqlite_ids.add(notion_id_str)

            # üî• FLATTEN NGAY T·ª™ ƒê·∫¶U: d√πng chung cho embedding + payload
            flat_row = {}
            for k, v in row_dict.items():
                flat_row[k] = flatten_recursive(v)


            text = build_embed_text(flat_row, table_name, mappings)

            # ‚úÖ Build payload t·ª´ d·ªØ li·ªáu ƒë√£ flatten (lo·∫°i b·ªè notion_id)
            payload = {}
            for k, v in flat_row.items():
                if k != "notion_id":
                    payload[k] = v

            payload["source_table"] = table_name
            # üî• NEW: G·∫Øn m√¥ t·∫£ b·∫£ng v√†o Qdrant
            description = get_table_description_from_sqlite(table_name)
            payload["table_description"] = description
            ids.append(notion_id_str)
            texts_to_embed.append(text)
            payloads.append(payload)

        # DEBUG: Print the first text to prove mapping works
        if texts_to_embed and total_synced == 0:
            print(f"  üëÄ [DEBUG] First text being embedded:\n     \"{texts_to_embed[0]}\"\n")

        if texts_to_embed:
            try:
                embeddings = embed_model.encode(
                    texts_to_embed, normalize_embeddings=True
                )

                for i, _id in enumerate(ids):
                    points_buffer.append(
                        PointStruct(
                            id=_id,
                            vector=embeddings[i].tolist(),
                            payload=payloads[i],
                        )
                    )

                if len(points_buffer) >= BATCH_SIZE:
                    client.upsert(
                        collection_name=GLOBAL_COLLECTION_NAME,
                        points=points_buffer,
                    )
                    total_synced += len(points_buffer)
                    print(f"  üíæ Upserted {len(points_buffer)} items...", end="\r")
                    points_buffer = []
                    gc.collect()

            except Exception as e:
                print(f"  ‚ö†Ô∏è Error embedding batch: {e}")

    if points_buffer:
        client.upsert(
            collection_name=GLOBAL_COLLECTION_NAME,
            points=points_buffer,
        )
        total_synced += len(points_buffer)

    conn.close()
    print(f"\n  ‚úÖ Finished upserting {total_synced} items from '{table_name}'")

    # 3. Xo√° c√°c ID "m·ªì c√¥i" trong Qdrant (c√≥ trong Qdrant nh∆∞ng kh√¥ng c√≤n trong SQLite)
    ids_to_delete = existing_ids - sqlite_ids
    if ids_to_delete:
        print(
            f"  üóëÔ∏è Deleting {len(ids_to_delete)} obsolete points in Qdrant for table '{table_name}'..."
        )
        try:
            client.delete(
                collection_name=GLOBAL_COLLECTION_NAME,
                points_selector=list(ids_to_delete),  # qdrant-client nh·∫≠n list id tr·ª±c ti·∫øp
            )
            print("  ‚úÖ Obsolete points deleted from Qdrant.")
        except Exception as e:
            print(f"  ‚ö†Ô∏è Error deleting obsolete points: {e}")
    else:
        print("  ‚úî No obsolete points to delete in Qdrant.")

    gc.collect()

def init_global_collection(client: QdrantClient):
    """Kh·ªüi t·∫°o collection duy nh·∫•t n·∫øu ch∆∞a c√≥"""
    try:
        if not client.collection_exists(GLOBAL_COLLECTION_NAME):
            print(f"Creating global collection: {GLOBAL_COLLECTION_NAME}")
            client.create_collection(
                collection_name=GLOBAL_COLLECTION_NAME,
                vectors_config=VectorParams(size=1024, distance=Distance.COSINE),
                check_compatibility=False
            )
            client.create_payload_index(
                collection_name=GLOBAL_COLLECTION_NAME,
                field_name="source_table",
                field_schema="keyword",
            )
            print("‚úÖ Collection & Index created.")
        else:
            print(f"Existing collection found: {GLOBAL_COLLECTION_NAME}")
    except Exception as e:
        print(f"Error checking/creating collection: {e}")


def create_dynamic_indexes(client: QdrantClient):
    """
    T·ª± ƒë·ªông t·∫°o index cho c√°c c·ªôt ƒë∆∞·ª£c d√πng l√†m dynamic filter (c·∫•u h√¨nh trong SQLite).
    """
    import json
    from qdrant_client.models import PayloadSchemaType

    print("Checking dynamic indexes...")
    try:
        conn = get_db_connection()
        cur = conn.cursor()
        cur.execute("SELECT dynamic_filters FROM collections_config WHERE enabled=1")
        rows = cur.fetchall()
        conn.close()

        for row in rows:
            raw = row[0]
            if not raw:
                continue
            try:
                config = json.loads(raw)
                target_col = config.get("target_col")
                if target_col:
                    print(f"  Creating index for dynamic filter column: {target_col}")
                    # M·∫∑c ƒë·ªãnh int cho ID, c√≥ th·ªÉ m·ªü r·ªông logic check type n·∫øu c·∫ßn
                    client.create_payload_index(
                        collection_name=GLOBAL_COLLECTION_NAME,
                        field_name=target_col,
                        field_schema=PayloadSchemaType.INTEGER,
                    )
            except Exception as e:
                print(f"  Warning processing filters: {e}")
                
    except Exception as e:
        print(f"Error creating dynamic indexes: {e}")



def cleanup_old_collections(client: QdrantClient):
    """
    Optional: X√≥a c√°c collections c≈© l·∫ª t·∫ª ƒë·ªÉ d·ªçn r√°c (n·∫øu tr∆∞·ªõc ƒë√¢y d√πng nhi·ªÅu collection)

    try:
        collections = client.get_collections().collections
        for c in collections:
            if c.name != GLOBAL_COLLECTION_NAME:
                print(f"üóëÔ∏è Deleting old fragmented collection: {c.name}")
                client.delete_collection(c.name)
    except Exception as e:
        print(f"Warning cleaning up: {e}")

def main():
    print("üöÄ Incremental Qdrant Sync Started...")

    # 1. Load Model
    print("üì¶ Loading embedding model...")
    try:
        embed_model = SentenceTransformer("BAAI/bge-m3", device="cpu")
    except Exception:
        embed_model = SentenceTransformer("keepitreal/vietnamese-sbert")

    # 2. Connect Qdrant
    print(f"üîó Connecting to Qdrant ({QDRANT_URL})...")
    client = QdrantClient(url=QDRANT_URL, api_key=QDRANT_API_KEY)

    # 3. Init Global Collection
    init_global_collection(client)
    create_dynamic_indexes(client)

    # 4. Optional: cleanup collections c≈©
    # cleanup_old_collections(client)

    # 5. Sync
    conn = get_db_connection()
    cur = conn.cursor()
    cur.execute("SELECT name FROM sqlite_master WHERE type='table' ORDER BY name")
    all_tables = [
        row[0] for row in cur.fetchall() if not row[0].startswith("sqlite_")
    ]
    conn.close()

    # N·∫øu script ƒë∆∞·ª£c g·ªçi v·ªõi t√™n b·∫£ng ‚Üí ch·ªâ sync b·∫£ng ƒë√≥
    specific_table = sys.argv[1] if len(sys.argv) > 1 else None

    if specific_table:
        print(f"üìå Running in single-table mode: {specific_table}")
        sync_table_to_global_collection(specific_table, embed_model, client)
    else:
        # Filter excluded tables
        filtered_tables = [t for t in all_tables if t not in EXCLUDED_TABLES]
        
        print(f"üìã Found {len(filtered_tables)} tables to sync (Excluded: {len(EXCLUDED_TABLES)}).")
        
        # 1. Sync valid tables
        for table in filtered_tables:
            sync_table_to_global_collection(table, embed_model, client)
            gc.collect()

        # 2. Cleanup excluded tables
        print("\nüßπ Cleaning up excluded tables from Qdrant...")
        for table in EXCLUDED_TABLES:
            delete_entire_table_from_qdrant(client, table)
            gc.collect()

    print("\nüéâ INCREMENTAL SYNC COMPLETED SUCCESSFULLY!")


if __name__ == "__main__":
    main()
